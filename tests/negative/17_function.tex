Test des appels de fonctions. Ici il y a plusieurs algorithmes TexSci. Le point d'entrée au programme est l'algorithme {\tt main}. Ici le résultat devrait être {\tt 120}.

\begin{texsci}{factorielle}
\Input{$n \in \Integer$}
\Output{$accu \in \Integer$}
\BlankLine
  $accu \leftarrow 1$\;
  \While{$n \neq 0$} {
    $accu \leftarrow n \times accu$\;
    $n \leftarrow n - 1$\;
  }
\end{texsci}

\begin{texsci}{main}
\Local{$fac \in \Integer$}
\BlankLine
$fac \leftarrow \mbox{factorielle($5$)}$\;
$\mbox{printInt($fac$)}$\;
\end{texsci}
